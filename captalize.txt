#!/bin/python3

import math
import os
import random
import re
import sys

# Complete the solve function below.
def solve(s):
    new_string = ""
    for i in range(len(s)):
        if s[i].isalpha and i == 0:
            new_string += s[i].upper()
        elif s[i].isalpha and s[i-1] == " ":
            new_string += s[i].upper()
        else:
            new_string += s[i]
    return new_string
if __name__ == '__main__':
    fptr = open(os.environ['OUTPUT_PATH'], 'w')

    s = input()

    result = solve(s)

    fptr.write(result + '\n')

    fptr.close()
